    wyckoff_dict = {}

    multiplicity_dict = {}
    coordinates_dict = {}

    for wpos, first_idx in wyckoff_dict.items():
        # Count multiplicity by counting occurrences
        multiplicity = wyckoffs.count(wpos)
        multiplicity_dict[wpos] = multiplicity

        # Get coordinates of first atom in this Wyckoff position
        coords = frac_coords[first_idx]
        coordinates_dict[wpos] = [
            to_fraction_string(coord) for coord in coords
